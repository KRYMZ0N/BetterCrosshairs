BetterCrosshairs Mod Development Documentation

This document covers the architectural design, mathematical foundation, and engine implementation details for replacing the default UI crosshair in Schedule I with a low-overhead, custom-compiled C++ spring physics model.

1. Physics Engine Architecture & ScalingThe mod replaces linear crosshair adjustments with a stateful system based on Hooke’s Law and Euler integration.Core MechanicsMass ($m$): $1.0\text{ f}$Spring Constant ($k$): $150.0\text{ f}$ 
(Controls the rate of restoration/snappiness)Damping Coefficient ($c$): $20.0\text{ f}$ (Prevents infinite harmonic oscillation)Scaling AdjustmentsBecause Unity world-space velocity vectors (meters per second) operate on a completely 
different scale than GUI rendering fields (pixels), a flat force multiplier is applied inside the C++ runtime to ensure visible displacement:Horizontal Strafing Scalar: 2000.0f (Calibrated to match the exact bounds of the game's 
default asset walking spread).Vertical/Fall Scalar: 1200.0f (Imposes a severe expansion penalty during jumps or vertical descents).ADS Modifier: 0.5f (Acts as a 50% force dampener when the Right Mouse Button is held, snapping the 
crosshair tightly inward).

2. Default HUD Deactivation StrategyData from the scene spatial scan reveals that the engine groups its default crosshair elements inside a parent container labeled Crosshair, alongside explicitly named component sub-nodes 
(Reticle, TopLine, BottomLine, LeftLine, RightLine).Rather than running heavy, recurring per-frame checks across the entire UI layer, the mod executes a precise target deactivation step inside the OnUpdate loop. 
This routine engages exclusively after the player object has moved away from the default world coordinate origin (0, 0, 0), preventing interference with loading layouts.

3. Production C# Implementation (Main.cs)This implementation replaces the diagnostic scanning logic. It targets the identified Crosshair and Reticle game objects directly to disable them, while continuing to pass 
separated structural movement vectors straight to the native CrosshairMath.dll runtime.
C#

using UnityEngine;
using System.IO;
using System.Reflection;
using System.Runtime.InteropServices;
using MelonLoader;
using Il2CppScheduleOne.DevUtilities;
using Il2CppScheduleOne.PlayerScripts;

[assembly: MelonInfo(typeof(BetterCrosshairs.Main), "Super Crosshair", "0.1.0", "KRYMZ0N")]
[assembly: MelonGame("TVGS", "Schedule I")]

namespace BetterCrosshairs;

public class Main : MelonMod {

    [DllImport("CrosshairMath", CallingConvention = CallingConvention.Cdecl)]
    public static extern float UpdateCrosshairPhysics(float speedXZ, float speedY, [MarshalAs(UnmanagedType.I1)] bool isFiring, [MarshalAs(UnmanagedType.I1)] bool isADS, float deltaTime);

    private Texture2D? _whiteTexture;
    private float _dynamicSpread = 0f;
    private bool _shouldDraw = false;
    private bool _defaultCrosshairHidden = false;
    
    // Cached references to eliminate loop allocations
    private Player? _cachedPlayer;
    private Vector3 _lastPlayerPosition;
    
    private const float CrosshairLength = 12f;
    private const float CrosshairThickness = 2f;
    private const float BaseGap = 6f;

    private Texture2D ReticleTexture {
        get {
            if (_whiteTexture == null) {
                _whiteTexture = Texture2D.whiteTexture;
            }
            return _whiteTexture;
        }
    }

    public override void OnInitializeMelon() {
        ExtractNativeDll();
        LoggerInstance.Msg("BetterCrosshairs Mod Successfully Initialized!");
    }

    public override void OnUpdate() {
        var playerInv = PlayerSingleton<PlayerInventory>.Instance;
        if (playerInv == null) {
            _shouldDraw = false;
            _cachedPlayer = null; 
            _defaultCrosshairHidden = false; // Reset tracking state across match boundaries
            return;
        }

        _shouldDraw = true;

        if (_cachedPlayer == null) {
            var allPlayers = UnityEngine.Object.FindObjectsOfType<Player>();
            foreach (var player in allPlayers) {
                if (player.IsOwner) { 
                    _cachedPlayer = player;
                    _lastPlayerPosition = _cachedPlayer.transform.position;
                    break; 
                }
            }
        }

        // Deactivate default HUD components once local player leaves the map origin point
        if (!_defaultCrosshairHidden && _cachedPlayer != null && _cachedPlayer.transform.position != Vector3.zero) {
            DisableDefaultHUD();
        }

        float speedXZ = 0.0f;
        float speedY = 0.0f;

        if (_cachedPlayer != null) {
            Vector3 currentPosition = _cachedPlayer.transform.position;
            
            if (Time.deltaTime > 0f) {
                // Isolate pure horizontal planar movement speed
                Vector3 currentXZ = new Vector3(currentPosition.x, 0, currentPosition.z);
                Vector3 lastXZ = new Vector3(_lastPlayerPosition.x, 0, _lastPlayerPosition.z);
                speedXZ = Vector3.Distance(currentXZ, lastXZ) / Time.deltaTime;

                // Isolate pure vertical acceleration speed
                speedY = Mathf.Abs(currentPosition.y - _lastPlayerPosition.y) / Time.deltaTime;
            }
            
            _lastPlayerPosition = currentPosition;
        }

        bool isPlayerFiring = Input.GetMouseButton(0); 
        bool isADS = Input.GetMouseButton(1);          
        
        // Execute dynamic math via native integration layer
        _dynamicSpread = UpdateCrosshairPhysics(speedXZ, speedY, isPlayerFiring, isADS, Time.deltaTime);
    }

    private void DisableDefaultHUD() {
        // Targets the primary structural UI container revealed by console dump
        GameObject defaultCrosshairContainer = GameObject.Find("Crosshair");
        if (defaultCrosshairContainer != null) {
            defaultCrosshairContainer.SetActive(false);
            _defaultCrosshairHidden = true;
            LoggerInstance.Msg("Successfully disabled default 'Crosshair' UI hierarchy.");
            return;
        }

        // Fail-safe redundancy targeting standalone reticle objects
        GameObject standaloneReticle = GameObject.Find("Reticle");
        if (standaloneReticle != null) {
            standaloneReticle.SetActive(false);
            _defaultCrosshairHidden = true;
            LoggerInstance.Msg("Successfully disabled standalone 'Reticle' UI asset.");
        }
    }

    public override void OnGUI() {
        if (!_shouldDraw) return;

        GUI.depth = 0;

        float width = Screen.width > 0 ? Screen.width : 1920f;
        float height = Screen.height > 0 ? Screen.height : 1080f;

        float centerX = width / 2f;
        float centerY = height / 2f;
        
        float totalGap = BaseGap + _dynamicSpread;

        GUI.color = Color.green;

        // Draw custom dynamic HUD layout
        GUI.DrawTexture(new Rect(centerX - totalGap - CrosshairLength, centerY - (CrosshairThickness / 2f), CrosshairLength, CrosshairThickness), ReticleTexture);
        GUI.DrawTexture(new Rect(centerX + totalGap, centerY - (CrosshairThickness / 2f), CrosshairLength, CrosshairThickness), ReticleTexture);
        GUI.DrawTexture(new Rect(centerX - (CrosshairThickness / 2f), centerY - totalGap - CrosshairLength, CrosshairThickness, CrosshairLength), ReticleTexture);
        GUI.DrawTexture(new Rect(centerX - (CrosshairThickness / 2f), centerY + totalGap, CrosshairThickness, CrosshairLength), ReticleTexture);
        
        GUI.color = Color.white;
    }

    private void ExtractNativeDll() {
        string targetPath = Path.Combine(Directory.GetCurrentDirectory(), "CrosshairMath.dll");
        var assembly = Assembly.GetExecutingAssembly();
        string resourceName = "BetterCrosshairs.CrosshairMath.dll";

        try {
            using (Stream? stream = assembly.GetManifestResourceStream(resourceName)) {
                if (stream == null) return;
                using (FileStream fileStream = new FileStream(targetPath, FileMode.Create, FileAccess.Write)) {
                    stream.CopyTo(fileStream);
                }
            }
        } 
        catch (System.IO.IOException) {
            LoggerInstance.Warning("CrosshairMath.dll target file is locked by the active process runtime.");
        }
    }
}
